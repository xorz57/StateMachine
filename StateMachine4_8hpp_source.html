64 state_machine_t(const state_t &state, transition_table_t<state_t, event_t, data_t> transition_table)

67 bool handle_event(const event_t &event, const data_t &data) {
68 const auto it = std::find_if(m_transition_table.begin(), m_transition_table.end(), [&](const transition_t<state_t, event_t, data_t> &transition) {
69 return transition.first.first == m_state && transition.first.second == event;
70 });
71 if (it != m_transition_table.end()) {
72 const auto &[guard, action, state] = it->second;
73 if (guard(data)) {
74 const auto it1 = m_leave_actions.find(m_state);
75 if (it1 != m_leave_actions.end()) {
76 it1->second(data);
77 }
78 m_state = state;
79 action(data);
80 const auto it2 = m_enter_actions.find(m_state);
81 if (it2 != m_enter_actions.end()) {
82 it2->second(data);
83 }
84 return true;
85 }
86 return true;
87 }
88 return false;
89 }
